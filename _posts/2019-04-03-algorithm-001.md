### 排序

```Java
class Sort{
    // insertion sort
    public static int[] insertionSort(int[] nums) {
        for (int i = 1; i < nums.length; i++) {
            int temp = nums[i];
            int j = i - 1;
            while (j >= 0 && nums[j] > temp) {
                nums[j + 1] = nums[j];
                j--;
            }
            nums[j + 1] = temp;
        }
        return nums;
    }

    // Bubble Sort
    public static int[] bubbleSort(int[] nums) {
        boolean flag = true;
        for (int i = 0; flag; i++) {
            flag = false;
            for (int j = nums.length - 1; j >= i + 1; j--) {
                if (nums[j] < nums[j - 1]) {
                    int temp = nums[j];
                    nums[j] = nums[j - 1];
                    nums[j - 1] = temp;
                    flag = true;
                }
            }
        }
        return nums;
    }
```
